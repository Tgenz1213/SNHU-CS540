# Clean and convert UZA SQ Miles
neighborhoods_df['UZA SQ Miles'] = pd.to_numeric(
    neighborhoods_df['UZA SQ Miles'].astype(str).str.replace(',', ''), 
    errors='coerce', downcast='float'
)

# Clean and convert Year/Month columns
for col in ['Last Closed Report Year', 'Last Closed FY End Month', 'Last Closed FY End Year']:
    neighborhoods_df[col] = pd.to_numeric(neighborhoods_df[col], errors='coerce', downcast='unsigned')

# Clean and convert Ridership/Distance columns
neighborhoods_df['Passenger Miles FY'] = pd.to_numeric(
    neighborhoods_df['Passenger Miles FY'].astype(str).str.replace(',', ''), 
    errors='coerce', downcast='unsigned'
)
neighborhoods_df['Unlinked Passenger Trips FY'] = pd.to_numeric(
    neighborhoods_df['Unlinked Passenger Trips FY'].astype(str).str.replace(',', ''), 
    errors='coerce', downcast='unsigned'
)

# Clean and convert Financial columns (removing both commas and dollar signs)
financial_cols = ['Fares FY', 'Operating Expenses FY', 'Avg Cost Per Trip FY', 'Avg Fares Per Trip FY']
for col in financial_cols:
    neighborhoods_df[col] = pd.to_numeric(
        neighborhoods_df[col].astype(str).str.replace(',', '').str.replace('$', ''), 
        errors='coerce', downcast='float'
    )
